function calculatePIT2026(brutto) { // 1. Складки социальные (ZUS) - 13.71% const emerytalna = brutto * 0.0976; const rentowa = brutto * 0.015; const chorobowa = brutto * 0.0245; const sumaZUS = emerytalna + rentowa + chorobowa; // 2. База для медицинской страховки const podstawaZdrowotna = brutto - sumaZUS; // 3. Складка zdrowotna - 9% (как в твоем PIT п. 122) const zdrowotna = podstawaZdrowotna * 0.09; // 4. Налог (PIT) с учетом налогового вычета (Kwota wolna 30k) // 300 - это ежемесячная сумма, уменьшающая налог (3600 / 12) const kosztyUzyskania = 250; // Стандартные расходы (п. 30 в твоем PIT) const podstawaOpodatkowania = Math.round(podstawaZdrowotna - kosztyUzyskania); let podatek = (podstawaOpodatkowania * 0.12) - 300; if (podatek < 0) podatek = 0; // 5. Итог: На руки const netto = brutto - sumaZUS - zdrowotna - podatek; return { netto: netto.toFixed(2), zus: sumaZUS.toFixed(2), tax: podatek.toFixed(2) }; }
